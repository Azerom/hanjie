
enum TailleResultats
{
	5x3,
	7x4,
	9x5,
	11x6
};

enum Difficulte
{
	FACILE,
	NORMAL,
	DIFFICILE
};

struct Cases
{
	int [][] grille;
	int hauteur;
	int largeur;
};

struct Niveau
{
	char *addresse;
	char *nom;
	enum TypeNiveau type;
	enum Difficulte difficulte;
};

struct Partie
{
	Cases pattern; //Solution du niveau
	Cases actuel; //Tableau remplis par l'utilisateur
	char [50] pseudo;
	TailleResultats tailleResultats;
	Niveau niveau;
	long temp;
	long date;
};

struct ElementHistorique
{
	long date;	
	char [50] pseudo;
	TypeNiveau type;
	Difficulte difficulte;
	long temp;
	Element *suivant;
}
